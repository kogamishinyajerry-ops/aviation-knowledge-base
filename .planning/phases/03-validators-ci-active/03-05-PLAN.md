```python
from typing import Iterator
```

```python
"""Validator regression tests.

Three test groups:
1. test_valid_corpus_passes — the entire valid tree must produce zero errors
   from ALL FIVE validators with the corpus_ctx index.
2. test_invalid_fixture_triggers_expected_rule — parametrised over every
   invalid fixture; asserts the rule named by the parent directory fires.
3. test_sanity_mutation_breaks_validation — proves the suite observes
   real behavior by mutating a valid record in-memory and asserting it
   becomes invalid.
"""
from __future__ import annotations
from pathlib import Path
import copy
import pytest

from scripts.validators import schema as v_schema
from scripts.validators import ids as v_ids
from scripts.validators import provenance as v_prov
from scripts.validators import relations as v_rel
from scripts.validators import links as v_links
from scripts.validators.errors import ValidationError


# Map of invalid-fixture directory name -> expected primary rule(s) the
# corresponding fixture must trigger. Sourced from
# tests/fixtures/invalid/README.md. Multiple rules listed = ALL must fire.
_INVALID_DIR_TO_RULES = {
    "missing-provenance": {"schema"},
    "missing-schema-version": {"schema"},
    "bad-uri-format": {"ids.uri-format"},          # schema layer also catches; we only assert the ids rule
    "bad-internal-id": {"ids.uri-format"},
    "h-darrieus-rejected": {"provenance.h-darrieus"},
    "h-darrieus-empty-reviewer": {"provenance.h-darrieus"},
    "pending-not-hybrid-reviewed": {"provenance.pending-gate"},
    "broken-source-ref": {"links.broken-source-ref"},
    "broken-relation-subject": {"relations.subject-not-found"},
    "broken-relation-object": {"relations.object-not-found"},
    "old-schema-version": {"provenance.schema-version-mismatch"},
    "empty-rationale": {"schema"},
}


def _run_all_validators(path: Path, record: dict, ctx: dict) -> list[ValidationError]:
    """Run every validator on the record and aggregate their ValidationErrors."""
    errors: list[ValidationError] = []
    errors.extend(v_schema.validate_record(path, record))
    errors.extend(v_ids.validate_record(path, record, **ctx))
    errors.extend(v_prov.validate_record(path, record, **ctx))
    errors.extend(v_rel.validate_record(path, record, **ctx))
    errors.extend(v_links.validate_record(path, record, **ctx))
    return errors


# -------------------------------------------------------------------------
# Group 1: valid corpus must pass cleanly
# -------------------------------------------------------------------------

@pytest.mark.valid_corpus
def test_valid_corpus_zero_errors(valid_records, corpus_ctx):
    """Every record under tests/fixtures/valid/ must produce zero error-severity
    ValidationErrors when run through all five validators."""
    failures: list[str] = []
    for path, record in valid_records:
        errs = _run_all_validators(path, record, corpus_ctx)
        errors = [e for e in errs if e.severity == "error"]
        if errors:
            failures.append(
                f"{path}:\n  " + "\n  ".join(e.format() for e in errors)
            )
    assert not failures, (
        f"{len(failures)} valid fixture(s) produced errors:\n\n"
        + "\n\n".join(failures)
    )


# -------------------------------------------------------------------------
# Group 2: each invalid fixture triggers its expected rule
# -------------------------------------------------------------------------

def _gather_invalid_fixture_params(invalid_dir: Path):
    """Walk tests/fixtures/invalid/ and yield (path, expected_rules) per fixture."""
    params = []
    for sub in sorted(invalid_dir.iterdir()):
        if not sub.is_dir():
            continue
        expected = _INVALID_DIR_TO_RULES.get(sub.name)
        if expected is None:
            # Skip directories we have not registered; but fail loud at collection.
            params.append(pytest.param(
                sub, set(), id=sub.name,
                marks=pytest.mark.xfail(reason=f"no rule mapping for dir '{sub.name}'"),
            ))
            continue
        for yaml_file in sorted(sub.rglob("*.y*ml")):
            params.append(pytest.param(yaml_file, expected, id=f"{sub.name}/{yaml_file.name}"))
    return params


@pytest.mark.invalid_corpus
@pytest.mark.parametrize(
    "fixture_path, expected_rules",
    _gather_invalid_fixture_params(
        Path(__file__).resolve().parent / "fixtures" / "invalid"
    ),
)
def test_invalid_fixture_triggers_expected_rule(
    fixture_path: Path, expected_rules: set[str], corpus_ctx
):
    """For each invalid fixture, assert the rule named by its parent directory fires."""
    from scripts.validators.loader import load_yaml
    record = load_yaml(fixture_path)
    errs = _run_all_validators(fixture_path, record, corpus_ctx)
    fired_rules = {e.rule for e in errs}
    missing = expected_rules - fired_rules
    assert not missing, (
        f"Fixture {fixture_path.name} expected rules {expected_rules} "
        f"but fired {fired_rules} (missing: {missing})"
    )


# -------------------------------------------------------------------------
# Group 3: sanity — mutation breaks validation
# -------------------------------------------------------------------------

@pytest.mark.sanity
def test_sanity_mutating_valid_record_fails(valid_records, corpus_ctx):
    """Pick a valid record, nuke its provenance block, confirm it now fails.
    Proves the test suite is actually observing validator output, not just
    blindly returning green."""
    assert valid_records, "no valid fixtures discovered — wave 1 may have failed"
    path, record = valid_records[0]
    mutated = copy.deepcopy(record)
    mutated.pop("provenance", None)
    errs = _run_all_validators(path, mutated, corpus_ctx)
    error_severity = [e for e in errs if e.severity == "error"]
    assert error_severity, (
        "removing provenance from a valid record should produce errors but "
        "the validator suite returned none — tests are not observing behavior."
    )


@pytest.mark.sanity
def test_sanity_h_darrieus_mutation_fires(valid_records, corpus_ctx):
    """Mutate a valid record into the H-Darrieus shape and confirm the rule fires."""
    assert valid_records
    path, record = valid_records[0]
    mutated = copy.deepcopy(record)
    mutated.setdefault("provenance", {})
    mutated["provenance"]["method"] = "ai_extracted"
    mutated["provenance"].pop("reviewer", None)
    mutated.setdefault("confidence", {})
    mutated["confidence"]["score"] = 0.99
    mutated["confidence"]["rationale"] = "synthetic mutation for sanity test purposes only."
    errs = _run_all_validators(path, mutated, corpus_ctx)
    rules = {e.rule for e in errs}
    assert "provenance.h-darrieus" in rules, (
        f"H-Darrieus mutation should fire provenance.h-darrieus; got rules: {rules}"
    )
```
