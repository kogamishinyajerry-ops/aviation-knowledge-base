"""Relations validator — cross-record subject/object resolution + supersession integrity.

This validator NEEDS the corpus-level index (ctx['by_id']) built by
validate.py so it can answer "does this URI exist as the id of any record?".
"""
from pathlib import Path
from typing import Any
from .errors import ValidationError


_RELATION_TYPES = {
    "part_of", "applicable_to", "constrained_by", "verified_by", "derived_from",
    "supersedes", "cites", "causes", "mitigated_by", "requires", "equivalent_to",
    "conflicts_with", "used_in", "interfaces_with", "complies_with",
    "applicable_during_phase",
}


def validate_record(path: Path, record: Any, **ctx) -> list[ValidationError]:
    errors: list[ValidationError] = []
    if not isinstance(record, dict):
        return errors
    by_id: dict = ctx.get("by_id") or {}
    file = str(path)
    rtype = record.get("type")

    # ---- A. relation subject/object resolution ---------------------------
    if rtype in _RELATION_TYPES:
        subj = record.get("subject")
        obj = record.get("object")
        if isinstance(subj, str) and subj not in by_id:
            errors.append(ValidationError(
                rule="relations.subject-not-found", severity="error", file=file,
                message=f"relation subject URI '{subj}' does not resolve to any record in the corpus",
                pointer="/subject",
            ))
        if isinstance(obj, str) and obj not in by_id:
            errors.append(ValidationError(
                rule="relations.object-not-found", severity="error", file=file,
                message=f"relation object URI '{obj}' does not resolve to any record in the corpus",
                pointer="/object",
            ))

    # ---- B. supersession integrity (entity-level) -------------------------
    # RegulationClause and similar entities can have status='superseded' +
    # superseded_by pointing at the active replacement.
    status = record.get("status")
    if status == "superseded":
        sb = record.get("superseded_by")
        if not isinstance(sb, str) or sb.strip() == "":
            errors.append(ValidationError(
                rule="relations.supersession-incomplete", severity="error", file=file,
                message="record has status='superseded' but no superseded_by URI",
                pointer="/superseded_by",
            ))
        elif sb not in by_id:
            errors.append(ValidationError(
                rule="relations.supersession-target-not-found", severity="error", file=file,
                message=f"superseded_by='{sb}' does not resolve to any record in the corpus",
                pointer="/superseded_by",
            ))

    return errors
```

"""Links validator — source.document_id resolution (PROV-06).

Verifies every record's source.document_id points at a record in the corpus
whose type=="Document". The schema layer already validates the URI format;
this layer validates the URI actually resolves AND resolves to a Document.
"""
from pathlib import Path
from typing import Any
from .errors import ValidationError


def validate_record(path: Path, record: Any, **ctx) -> list[ValidationError]:
    errors: list[ValidationError] = []
    if not isinstance(record, dict):
        return errors
    by_id: dict = ctx.get("by_id") or {}
    file = str(path)
    src = record.get("source")
    if not isinstance(src, dict):
        return errors  # schema layer handles missing-source-when-required

    doc_id = src.get("document_id")
    if not isinstance(doc_id, str) or doc_id.strip() == "":
        return errors  # schema layer handles this

    target = by_id.get(doc_id)
    if target is None:
        errors.append(ValidationError(
            rule="links.broken-source-ref", severity="error", file=file,
            message=(
                f"source.document_id='{doc_id}' does not resolve to any record "
                "in the corpus. Either add the Document entity or correct the URI."
            ),
            pointer="/source/document_id",
        ))
        return errors

    _, target_record = target
    if isinstance(target_record, dict) and target_record.get("type") != "Document":
        errors.append(ValidationError(
            rule="links.source-not-document", severity="error", file=file,
            message=(
                f"source.document_id='{doc_id}' resolves to a record of type="
                f"'{target_record.get('type')}', expected 'Document'. PROV-06 "
                "requires source citations to point at Document entities only."
            ),
            pointer="/source/document_id",
        ))

    return errors
```
